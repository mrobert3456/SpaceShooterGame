202  private void Win_KeyDown(object sender, KeyEventArgs e)
203  {
204  switch (e.Key)
205  {
206  case Key.W: this.movey = 15; break;
207  case Key.A: this.movex = 15; break;
208  case Key.D: this.movex = -15; break;
209  case Key.S: this.movey = -15; break;
210  case Key.F:
211  {
212  IGameModel newModel = this.logic.NextLevel();
213  if (this.model != newModel)
214  {
215  this.model = newModel;
216  GameControl.Level = 2;
217  this.shootLogic = LogicFactory.CreateShootLogic(this.model);
218  this.logic = LogicFactory.CreateMainLogic(this.model);
219  this.renderer = RenderFactory.CreateRender(this.model);
220  this.filelogic = LogicFactory.CreateFileLogic(this.model);
221  this.win.Background = new ImageBrush(this.nextlevel);
222  }
223  }
224 
225  break;
226  case Key.Escape:
227  {
228  this.enemyShootT.Stop();
229  this.dt.Stop();
230  this.PauseMenu();
231  }
232 
233  break;
